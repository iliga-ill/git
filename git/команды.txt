git branch "name" (создание класса)
git checkout C6(move HEAD)
git branch -f master(class) C6(HEAD) (перенос класса к хеду)
git branch -f master bugFix (перенос 1 класса к другому)
git reset(предыдущая ветка перенос, эта исчезает(блекнет) )
git revert(делает новый коммит с противоположными изменениями, чем у предыдущего(удаляет изменения))
git cherry-pick <Commit1> <Commit2> <...> (копируем коммиты на место хеда)
rebase - сливает 2 коммита в 1, обьединяя изменения в них.
git rebase -i HEAD~n (все коммиты в списке всплывают, выбираем, какие копировать, и они копируются после n,
 можно перетаскивать последовательность)
git rebase HEAD bugFix (копируем после хеда то, что в багфикс(коммиты) вместе с классом)
git rebase HEAD С6 (копируем после хеда то, что в багфикс(коммиты) без класса)
git tag v1(тэг) C1(коммит) (создание тегов)
git describe <ref>(показывает как далеко находится ближайший тег)
git checkout master^
git merge master - сливаем текущую ветку с мастером

git clone - копирование созданного на удаленный репозиторий
git checkout o/master; git commit - две команды в строчку?
git fetch - копирование(скачивание) данных из удаленного репозитория
git pull - скачивает данные с рипозитория и сразу сливает  с вашими
git fakeTeamwork foo 3(типо изменение удаленного репозитория кем-то, foo - новый/не новый класс, 3-кол-во коммитов)
git push o отвечает за загрузку ваших изменений в указанный удалённый репозиторий
git pull --rebase - скачивает данные с удаленного репозитория и обьединяет с вашими
git checkout -b totallyNotMaster o/master - создание новой ветки не мастер и команда ей следить за мастером
после этого использование команды  git push/pull будет копировать коммиты из не мастера в удаленного мастера
git branch -u o/master foo то же самое, что и пред команда
git push origin master - конкретно передаем то, что в мастере в удаленный origin
git push origin <источник>:<получатель>
git fetch origin foo
git fetch origin <источник>:<получатель>
git push origin :side стирает верку в удаленном респозитории путем передавания в него ничего
git fetch origin :bar добавляет новый класс в удаленном репозитории
git pull origin master:foo
